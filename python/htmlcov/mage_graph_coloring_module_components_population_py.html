166 def _calculate_correlation(self, first: Individual, second: Individual) -> float: 

167 correlation = 0 

168 for node_1 in self._graph.nodes: 

169 for node_2 in range(node_1 + 1, len(self._graph)): 

170 S_first = -1 if first[node_1] == first[node_2] else 1 

171 S_second = -1 if second[node_1] == second[node_2] else 1 

172 correlation += S_first * S_second 

173 return correlation

192 self._sum_conflicts_weight -= old_indv.conflicts_weight
